cmake_minimum_required(VERSION 3.10)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

project(voxil VERSION 1.0)
file(GLOB_RECURSE SRC_FILES ../src/*.cpp)
add_executable(voxil ${SRC_FILES})
target_link_libraries(voxil PUBLIC sfml-graphics)
target_link_libraries(voxil PUBLIC sfml-window)
target_link_libraries(voxil PUBLIC sfml-system)
